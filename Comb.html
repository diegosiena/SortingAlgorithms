void insertionSort(int numeros[], int tam) {
    int i, j, eleito;
    for (i = 1; i < tam; i++){
        eleito = numeros[i];
        j = i - 1;
        while ((j>=0) && (eleito < numeros[j])) {
            numeros[j+1] = numeros[j];
            j--;
        }
        numeros[j+1] = eleito;
    }
}
Javascript
function insertionSort(array) {
    var len = array.length,
        value,
        i,
        j;
    
    for (i = 0; i < len; i++) {
        value = array[i];
        
        for (j = i - 1; j > -1 && array[j] > value; j--) {
            array[j + 1] = array[j];
        }
        array[j + 1] = value;
    }
    
    return array;
}
