Linguagem C
void selection_sort(int num[], int tam) { 

    int i, j, min, aux;
    for (i = 0; i < (tam-1); i++) {

        min = i;
        for (j = (i+1); j < tam; j++) {

            if(num[j] < num[min]) {
                min = j;
            }
        }

        if (i != min) {
            aux = num[i];
            num[i] = num[min];
            num[min] = aux;
        }
    }
}

function selectionSort(array) {

    var len = array.length,
        min;

    for (i=0; i < len; i++){

        min = i;
        for (j=i+1; j < len; j++){

            if (array[j] < array[min]) {
                min = j;
            }
        }

        if (i != min) {
            var temp = array[i];
            array[i] = array[min];
            array[min] = temp;
        }
    }

    return array;
}
